# ERD Mode Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Add an ERD mode to the existing single-page schema generator, with a `Schema | ERD` mode switch in the header. The ERD mode draws Elmasri/Navathe-style diagrams (entities, attributes, relationships) matching the reference picture used during brainstorming.

**Architecture:** Keep everything in the single `index.html`. Reshape `state` into `{ mode, schema, erd }` so each mode owns its data. Header chrome (sample/reset/PNG/PDF) dispatches by mode. The ERD builder mirrors the existing table-card UI pattern. The ERD renderer is a new function set that reuses the existing `mk`/`txt`/drag/export pipeline.

**Tech Stack:** Vanilla JS, inline SVG, no build step, no test framework. Verification is **manual in-browser** — each task lists exact click-paths and observable outcomes.

**Verification convention:** Every task ends with a manual verification step. "Open the page" means open `index.html` in a browser (the user uses a local file or static server — either works since there's no build step). "DevTools console clean" means no errors or warnings in the browser console after the action.

**Spec:** `docs/superpowers/specs/2026-05-17-erd-mode-design.md`

## Task 4: Make the SVG drag handler mode-aware

**Files:**
- Modify: `index.html` (the `wireDrag` IIFE near line 952)

**Why:** Schema mode uses `data-table-drag="<tid>"` on the table group. ERD mode will use `data-erd-drag="<kind>:<id>"` on entities, relationships, and attribute ovals. We generalize the handler so both work.

- [ ] **Step 1: Replace the body of `wireDrag`**

Find the existing IIFE near line 952:

```js

  let drag = null;
  // ...existing implementation...
})();
```

Replace the entire IIFE with:

```js
(function wireDrag() {
  const svgEl = document.getElementById('svg');
  let drag = null;
  const toSvg = (evt) => {
    const r = svgEl.getBoundingClientRect();
    return { x: evt.clientX - r.left, y: evt.clientY - r.top };
  };

  // Returns { ref, pos } for the schema/erd node under the pointer, or null.
  function findDragTarget(e) {
    // Schema mode: table drag.
    const schHandle = e.target.closest('[data-table-drag]');
    if (schHandle && state.mode === 'schema') {
      const tid = schHandle.getAttribute('data-table-drag');
      const t = state.schema.tables.find((tt) => tt.id === tid);
      if (!t) return null;
      const pos = layoutTables()[tid];
      return { ref: t, pos };
    }
    // ERD mode: entity / relationship / attribute drag.
    const erdHandle = e.target.closest('[data-erd-drag]');
    if (erdHandle && state.mode === 'erd') {
      const key = erdHandle.getAttribute('data-erd-drag'); // "<kind>:<id>"
      const [kind, id] = key.split(':');
      const target = findErdNode(kind, id);
      if (!target) return null;
      const positions = layoutErd();
      const pos = positions[key];
      if (!pos) return null;
      return { ref: target, pos };
    }
    return null;
  }

  svgEl.addEventListener('pointerdown', (e) => {
    const t = findDragTarget(e);
    if (!t) return;
    const pt = toSvg(e);
    drag = { ref: t.ref, offX: pt.x - t.pos.x, offY: pt.y - t.pos.y, pointerId: e.pointerId };
    document.body.style.userSelect = 'none';
    try { svgEl.setPointerCapture(e.pointerId); } catch (_) {}
    e.preventDefault();
  });
  document.addEventListener('pointermove', (e) => {
    if (!drag) return;
    const pt = toSvg(e);
    drag.ref.x = Math.max(0, pt.x - drag.offX);
    drag.ref.y = Math.max(0, pt.y - drag.offY);
    if (state.mode === 'schema') renderSchemaSvg();
    else renderErdSvg();
  });
  const endDrag = (e) => {
    if (!drag) return;
    try { if (e && e.pointerId != null) svgEl.releasePointerCapture(e.pointerId); } catch (_) {}
    drag = null;
    document.body.style.userSelect = '';
  };
  document.addEventListener('pointerup', endDrag);
  document.addEventListener('pointercancel', endDrag);
})();
```

// ============================================================================
// ERD MODE: data factories, sample, lookups

function makeAttribute(overrides = {}) {
  return {
    id: uid('a'),
    name: 'attribute',
    kind: 'simple',           // 'simple' | 'key' | 'partial-key' | 'composite' | 'multivalued' | 'derived'
    children: [],             // only used for kind === 'composite'
    ...overrides,
  };
}

function makeEntity(overrides = {}) {
  const n = state.erd.entities.length + 1;
  return {
    id: uid('e'),
    name: `ENTITY_${n}`,
    kind: 'strong',           // 'strong' | 'weak'
    attributes: [],
    ...overrides,
  };
}

function makeParticipant(overrides = {}) {
  return {
    id: uid('p'),
    entityId: '',
    role: '',
    cardinality: '1',         // '1' | 'N' | 'M'
    participation: 'partial', // 'total' | 'partial'
    ...overrides,
  };
}

function makeRelationship(overrides = {}) {
  const n = state.erd.relationships.length + 1;
  return {
    id: uid('R'),
    name: `RELATIONSHIP_${n}`,
    kind: 'normal',           // 'normal' | 'identifying'
    attributes: [],
    participants: [],
    ...overrides,
  };
}

function findErdNode(kind, id) {
  if (kind === 'entity') return state.erd.entities.find((e) => e.id === id) || null;
  if (kind === 'relationship') return state.erd.relationships.find((r) => r.id === id) || null;
  if (kind === 'attribute') {
    const search = (list) => {
      for (const a of list) {
        if (a.id === id) return a;
        const inner = search(a.children || []);
        if (inner) return inner;
      }
      return null;
    };
    for (const e of state.erd.entities) {
      const a = search(e.attributes || []);
      if (a) return a;
    }
    for (const r of state.erd.relationships) {
      const a = search(r.attributes || []);
      if (a) return a;
    }
    return null;
  }
  return null;
}

function clearAttrPin(a) {
  delete a.x; delete a.y;
  (a.children || []).forEach(clearAttrPin);
}

    ]}),
    makeAttribute({ name: 'Bdate', kind: 'simple' }),
    makeAttribute({ name: 'Address', kind: 'simple' }),
    makeAttribute({ name: 'Sex', kind: 'simple' }),
    makeAttribute({ name: 'Salary', kind: 'simple' }),
  ]});
  const department = makeEntity({ name: 'DEPARTMENT', kind: 'strong', attributes: [
    makeAttribute({ name: 'Name', kind: 'key' }),
    makeAttribute({ name: 'Number', kind: 'key' }),
    makeAttribute({ name: 'Locations', kind: 'multivalued' }),
    makeAttribute({ name: 'Number_of_employees', kind: 'derived' }),
  ]});
  const project = makeEntity({ name: 'PROJECT', kind: 'strong', attributes: [

function wireErdBuilderEvents() {
  // === Helpers ===
  const findEntity = (el) => state.erd.entities.find((e) => e.id === el.closest('.e-card').dataset.eid);
  const findRel = (el) => state.erd.relationships.find((r) => r.id === el.closest('.R-card').dataset.rid);
  const findAttrAndOwner = (el) => {
    // Walks up the DOM to find the attribute row + its owner (entity or relationship)
    // + whether the attribute is a sub-attribute (nested under a composite parent).
    const row = el.closest('.attr-row');
    const aid = row.dataset.aid;
    const composite = el.closest('[data-attr-host="composite"]');
    if (composite) {
      const parentAid = composite.dataset.parentAid;
      const ownerCard = el.closest('.e-card');         // composites only on entities
      const entity = state.erd.entities.find((e) => e.id === ownerCard.dataset.eid);
      const parent = findAttrInList(entity.attributes, parentAid);
      const attr = (parent.children || []).find((c) => c.id === aid);
      return { owner: entity, ownerKind: 'entity', parent, attr };
    }
    if (el.closest('.e-card')) {
      const entity = state.erd.entities.find((e) => e.id === el.closest('.e-card').dataset.eid);
      const attr = findAttrInList(entity.attributes, aid);
      return { owner: entity, ownerKind: 'entity', parent: null, attr };
    }
    if (el.closest('.R-card')) {
      const rel = state.erd.relationships.find((r) => r.id === el.closest('.R-card').dataset.rid);
      const attr = findAttrInList(rel.attributes, aid);
      return { owner: rel, ownerKind: 'relationship', parent: null, attr };

    return null;

  g.appendChild(mk('ellipse', ovalAttrs));
  if (doubled) {
    g.appendChild(mk('ellipse', { cx: pos.x, cy: pos.y, rx: rx - ins, ry: ry - ins, fill: 'none', stroke: ERD_C.ATTR_STROKE, 'stroke-width': ERD_DIMS.LINE_W }));
  }
  // Label.
  const label = a.name || '(unnamed)';

    'font-size': 11, 'font-weight': a.kind === 'key' || a.kind === 'partial-key' ? 600 : 400,
  });

  // Underline for key / partial-key — drawn as an explicit <line> so it survives PNG/PDF rasterization.
  if (a.kind === 'key' || a.kind === 'partial-key') {
    // Approximate text width: 6.6px per char at font-size 11 (JetBrains Mono). Sufficient for the underline length.
    const halfW = (label.length * 6.6) / 2;
    const yLine = pos.y + ry / 2 + 1;

    g.appendChild(mk('line', lineAttrs));

// Single or double connecting line between two points. Used for entity↔relationship and attribute↔owner.
function drawErdConnector(svg, x1, y1, x2, y2, doubled) {
  const stroke = ERD_C.LINE;
  if (!doubled) {
    svg.appendChild(mk('line', { x1, y1, x2, y2, stroke, 'stroke-width': ERD_DIMS.LINE_W }));
    return;
  }
  // Double line: offset perpendicular to (dx, dy) by ± DBL_LINE_GAP/2.
  const dx = x2 - x1, dy = y2 - y1;
  const len = Math.max(1, Math.hypot(dx, dy));
  const ox = -dy / len * (ERD_DIMS.DBL_LINE_GAP / 2);
  const oy =  dx / len * (ERD_DIMS.DBL_LINE_GAP / 2);
  svg.appendChild(mk('line', { x1: x1 + ox, y1: y1 + oy, x2: x2 + ox, y2: y2 + oy, stroke, 'stroke-width': ERD_DIMS.LINE_W }));
  svg.appendChild(mk('line', { x1: x1 - ox, y1: y1 - oy, x2: x2 - ox, y2: y2 - oy, stroke, 'stroke-width': ERD_DIMS.LINE_W }));
}

// Computes positions keyed by "<kind>:<id>" for every entity, relationship, and attribute.
// Auto-layout is overridden by any pinned x/y stored on the data object (set via drag).
function layoutErd() {
  const positions = {};
  const ENT_COLS = 3;
  const ENT_GAP_X = 280, ENT_GAP_Y = 280, ENT_X0 = 180, ENT_Y0 = 120;

  // 1. Entities in a grid.
  state.erd.entities.forEach((e, i) => {
    const col = i % ENT_COLS, row = Math.floor(i / ENT_COLS);
    const auto = { x: ENT_X0 + col * ENT_GAP_X, y: ENT_Y0 + row * ENT_GAP_Y };
    const pinned = typeof e.x === 'number' && typeof e.y === 'number';
    positions[`entity:${e.id}`] = pinned ? { x: e.x, y: e.y } : auto;
  });

  // 2. Attribute fan layout — for each entity, place attribute ovals on a ring above the entity.
  //    Skips pinned attributes.
  state.erd.entities.forEach((e) => {
    const center = positions[`entity:${e.id}`];
    const attrs = e.attributes || [];
    const R = 110; // ring radius from entity centre
    // Fan above: angles range from 200deg to 340deg (i.e. above the entity, sweeping right→left).
    // For evenly distributing N attrs we map index → angle.
    attrs.forEach((a, i) => {
      const pinned = typeof a.x === 'number' && typeof a.y === 'number';
      if (pinned) {
        positions[`attribute:${a.id}`] = { x: a.x, y: a.y };
      } else {
        const n = attrs.length;
        const tStart = -Math.PI * 1.0,  tEnd = -Math.PI * 0.05; // sweep through top
        const t = n === 1 ? (tStart + tEnd) / 2 : tStart + (tEnd - tStart) * (i / (n - 1));
        positions[`attribute:${a.id}`] = { x: center.x + R * Math.cos(t), y: center.y + R * Math.sin(t) };
      }
      // Composite children — fan above the parent.
      (a.children || []).forEach((c, j) => {
        const pPos = positions[`attribute:${a.id}`];
        const pinnedC = typeof c.x === 'number' && typeof c.y === 'number';
        if (pinnedC) {
          positions[`attribute:${c.id}`] = { x: c.x, y: c.y };
          return;
        }
        const m = (a.children || []).length;
        const ang = -Math.PI / 2 + (j - (m - 1) / 2) * 0.5;
        const r = 80;
        positions[`attribute:${c.id}`] = { x: pPos.x + r * Math.cos(ang), y: pPos.y + r * Math.sin(ang) };
      });
    });
  });

  // 3. Relationships — centroid of participants. Self-relationships get an offset.
  state.erd.relationships.forEach((r) => {
    const pinned = typeof r.x === 'number' && typeof r.y === 'number';
    if (pinned) {
      positions[`relationship:${r.id}`] = { x: r.x, y: r.y };
    } else {
      const pts = (r.participants || [])
        .map((p) => positions[`entity:${p.entityId}`])
        .filter(Boolean);
      if (pts.length === 0) {
        positions[`relationship:${r.id}`] = { x: ENT_X0, y: ENT_Y0 - 200 };
      } else if (pts.length === 1 || (pts.length === 2 && pts[0] === pts[1])) {
        // Unary — offset to the left of the entity.
        positions[`relationship:${r.id}`] = { x: pts[0].x - 200, y: pts[0].y };
      } else {
        const cx = pts.reduce((s, p) => s + p.x, 0) / pts.length;
        const cy = pts.reduce((s, p) => s + p.y, 0) / pts.length;
        positions[`relationship:${r.id}`] = { x: cx, y: cy };
      }
    }
    // Relationship attributes — fan above the diamond.
    const center = positions[`relationship:${r.id}`];
    (r.attributes || []).forEach((a, i) => {
      const pinned = typeof a.x === 'number' && typeof a.y === 'number';
      if (pinned) {
        positions[`attribute:${a.id}`] = { x: a.x, y: a.y };
      } else {
        const n = (r.attributes || []).length;
        const ang = -Math.PI / 2 + (i - (n - 1) / 2) * 0.5;
        const R = 80;
        positions[`attribute:${a.id}`] = { x: center.x + R * Math.cos(ang), y: center.y + R * Math.sin(ang) };
      }
    });
  });

  return positions;
}

function connectAttributeChain(svg, a, ownerPos, positions) {

  });
